DATA TABLE
==========

PROPERTIES
----------
columns: var = []                 // [{key, label, width?, sortable?, align?}]
rows: var = []
showCheckboxes: bool = false
sortable: bool = true
sortColumn: string = ""
sortAscending: bool = true
selectedRows: var = []
stickyHeader: bool = true
striped: bool = false
hoverable: bool = true
bordered: bool = false

SIGNALS
-------
rowClicked(int index, var row)
rowDoubleClicked(int index, var row)
selectionChanged(var selectedRows)
sortChanged(string column, bool ascending)

USAGE
-----
import SmartUIBeta

// Basic table
DataTable {
    columns: [
        {key: "name", label: "Name"},
        {key: "email", label: "Email"},
        {key: "role", label: "Role"}
    ]
    
    rows: [
        {name: "John", email: "john@example.com", role: "Admin"},
        {name: "Jane", email: "jane@example.com", role: "User"}
    ]
    
    onRowClicked: (index, row) => {
        console.log("Clicked:", row.name)
    }
}
